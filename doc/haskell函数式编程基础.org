*** quickCheck

max :: Integer -> Integer -> Integer
max x y
  | x >= y      = x
  | otherwise   = y

-- An alternative definition of max which uses if ... then ... else ...

max' :: Integer -> Integer -> Integer
max' x y
  = if x >= y then x else y

prop_compareMax :: Integer -> Integer -> Bool
prop_compareMax x y =
    max x y == max' x y

ghci> quickCheck prop_compareMax
